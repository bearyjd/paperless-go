- [ ] **Step 3: Handle the action**

In the action handler, add:

```dart
case 'redo_ocr':
  final confirmed = await showDialog<bool>(
    context: context,
    builder: (ctx) => AlertDialog(
      title: const Text('Re-run OCR?'),
      content: Text('Re-run OCR on ${selectedIds.length} document${selectedIds.length == 1 ? '' : 's'}? This may take a while.'),
      actions: [
        TextButton(
          onPressed: () => Navigator.pop(ctx, false),
          child: const Text('Cancel'),
        ),
        FilledButton(
          onPressed: () => Navigator.pop(ctx, true),
          child: const Text('Re-run'),
        ),
      ],
    ),
  );
  if (confirmed != true || !context.mounted) return;
  try {
    await api.bulkEdit(
      documents: selectedIds.toList(),
      method: 'redo_ocr',
    );
    onClearSelection();
    if (context.mounted) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('OCR re-run started for ${selectedIds.length} document${selectedIds.length == 1 ? '' : 's'}')),
      );
    }
  } catch (e) {
    if (context.mounted) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('Failed: $e')),
      );
    }
  }
```

Adapt variable names (`selectedIds`, `api`, `onClearSelection`) to match the actual code in the file.
